const schemas = JSON.parse(pm.variables.get('schemas'));
pm.test('200 OK', () => pm.response.to.have.status(200));
pm.test('Content-Type is application/json', () => pm.expect(pm.response.headers.get('Content-Type') || '').to.include('application/json'));
const v = tv4.validate(pm.response.json(), schemas.status, true, true);
pm.test('strict status schema (no extra fields)', () => pm.expect(v, tv4.error && tv4.error.message).to.be.true);
